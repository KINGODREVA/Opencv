import cv2

img = cv2.imread('../img/blank_500.jpg)

cv2.rectangle(img, (450, 200), (200,450), (0,0,255), -1)
cv2.putImage(img, "wonjung", (100,200), cv2.FONT_HERSHEY_TRIPLEX,2, (0,0,0))

cv2.imshow('rectangle', img)
cv2.waitKey(0)
cv2.destroyAllWindows()
